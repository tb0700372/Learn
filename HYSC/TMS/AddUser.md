DECLARE @FromName VARCHAR(20), @UserID VARCHAR(20),@UserName VARCHAR(20)
SELECT @UserID='shenhuayan',@UserName='沈华燕',@FromName='柴丽丽'
INSERT dbo.SEC_USER
(
  [USERID], [CLIENTID], [USERTYPE], [COMMISSIONRATE], [PARENTID], [DEPTID], [CNAME], [ENAME]
  , [TITLE], [PWDWEB], [SEX], [PWDINNER], [BIRTHDAY], [DISCOUNT], [PHONE], [STAFFSTATUS], [MP]
  , [SUPERERNID], [BP], [EMPLOYEETYPE], [EMAIL], [ACTIVE], [POSTALCODE], [AREAID], [ADDRESS]
  , [REMARK], [CREATEDATE], [CREATEBY], [EDITDATE], [EDITBY], [HIREDATE], [FIREDATE], [DOCUMENTNO]
  , [STDRATE], [IMAGE], [GRADUATE], [BACKGROUND], [EXPERIENCE], [WHSID], [OWNCLIENTID], [ADDRESS1]
  , [ADDRESS2], [ADDRESS3], [ADDRESS4], [CITYID], [STATEID], [COUNTRYID], [ZIP], [CONTACT1], [PHONE1]
  , [MOBILE1], [DISTRICTID], [TOKEN], [LANGUAGE], [CUSTOMERID]
)

[USERID]=@UserID, [CLIENTID], [USERTYPE], [COMMISSIONRATE], [PARENTID], [DEPTID], [CNAME]=@UserName, [ENAME]=@UserID
, [TITLE], [PWDWEB], [SEX], [PWDINNER], [BIRTHDAY], [DISCOUNT], [PHONE], [STAFFSTATUS], [MP]
, [SUPERERNID], [BP], [EMPLOYEETYPE], [EMAIL], [ACTIVE], [POSTALCODE], [AREAID], [ADDRESS]
, [REMARK], [CREATEDATE], [CREATEBY], [EDITDATE], [EDITBY], [HIREDATE], [FIREDATE], [DOCUMENTNO]
, [STDRATE], [IMAGE], [GRADUATE], [BACKGROUND], [EXPERIENCE], [WHSID], [OWNCLIENTID], [ADDRESS1]
, [ADDRESS2], [ADDRESS3], [ADDRESS4], [CITYID], [STATEID], [COUNTRYID], [ZIP], [CONTACT1], [PHONE1]
, [MOBILE1], [DISTRICTID], [TOKEN], [LANGUAGE], [CUSTOMERID]
FROM
SEC_USER where CNAME=@FromName

select USERID,COUNT(USERID) from SEC_USER group by USERID

DECLARE @FromName VARCHAR(20), @UserID VARCHAR(20),@UserName VARCHAR(20)

SELECT @UserID='shenhuayan',@UserName='沈华燕',@FromName='徐金花'

(
ERN,ROLE,UPDATE_DATE,UPDATE_BY
)
select ERN=@UserID,ROLE,GETDATE(),'Teng' from SEC_USERROLE
where ERN=(
  select USERID from SEC_USER where CNAME=@FromName
  )
